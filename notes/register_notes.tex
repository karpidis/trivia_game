% ──────────────────────────────────────────────
\section{How Users Are Stored}

Passwords are saved in plain text in a \texttt{.txt} file.\\
One user per line, username and password separated by a colon:

\begin{lstlisting}
andreas:mypassword
john:1234
\end{lstlisting}

File location: \texttt{app/users.txt} (created automatically on first register)

% ──────────────────────────────────────────────
\section{The Code --- app/auth.py}

\begin{lstlisting}
from flask import Blueprint, request, session, redirect, url_for, render_template
import os

auth_bp = Blueprint("auth", __name__)

USERS_FILE = os.path.join(os.path.dirname(__file__), "users.txt")

def load_users():
    users = {}
    if os.path.exists(USERS_FILE):
        with open(USERS_FILE, "r") as f:
            for line in f:
                line = line.strip()
                if ":" in line:
                    username, password = line.split(":", 1)
                    users[username] = password
    return users

@auth_bp.route("/register", methods=["GET", "POST"])
def register():
    if request.method == "POST":
        username = request.form.get("username").strip()
        password = request.form.get("password").strip()

        users = load_users()

        if username in users:
            return render_template("register.html",
                                   error="Username already exists.")

        with open(USERS_FILE, "a") as f:
            f.write(f"{username}:{password}\n")

        return redirect(url_for("auth.login"))

    return render_template("register.html")
\end{lstlisting}

\section{What Each Part Does}

\begin{center}
\begin{tabular}{ll}
\toprule
\textbf{Part} & \textbf{Why} \\
\midrule
\texttt{load\_users()} & Reads the txt file into a dictionary \\
                        & so you can check if a username exists \\
\addlinespace
\texttt{if username in users} & Prevents duplicate usernames \\
\addlinespace
\texttt{open(..., "a")} & \texttt{"a"} = append mode --- adds the new user \\
                         & without deleting the others \\
\addlinespace
\texttt{split(":", 1)} & The \texttt{1} means split only on the first colon \\
                        & so passwords containing \texttt{:} won't break it \\
\addlinespace
\texttt{redirect(url\_for(...))} & After registering, send the user to login \\
                                  & Never render directly after a form submit \\
\bottomrule
\end{tabular}
\end{center}

% ──────────────────────────────────────────────
\section{The Register Flow Step by Step}

\begin{enumerate}
    \item User visits \texttt{/register} $\rightarrow$ GET request $\rightarrow$ show the form
    \item User fills in username and password, submits $\rightarrow$ POST request
    \item Load all existing users from \texttt{users.txt}
    \item Check if the username already exists
    \begin{itemize}
        \item If yes $\rightarrow$ show the form again with an error message
        \item If no $\rightarrow$ continue
    \end{itemize}
    \item Append \texttt{username:password} as a new line in \texttt{users.txt}
    \item Redirect to \texttt{/login}
\end{enumerate}
